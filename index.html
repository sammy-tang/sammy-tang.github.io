        int poTwo1 = twoTurnTen(polynthic1);
        int poTwo2 = twoTurnTen(polynthic2);
        int m = log[poTwo1];
        int n = log[poTwo2];
        int ans = exp[(m + n) % 255];
        // 十进制转二进制
        return tenTurnTwo(ans);

        int tmp = 1;
        int[] birth = null;
        int[] divisor = { 1, 1, 0, 0, 0, 1, 1, 0, 1 };
        int i = 0;
        while (true) {
            birth = new int[i + 1];
            for (int j : birth) {
                j = 0;
            }
            birth[i] = 1;
            if (i == 8) {
                tmp = 99;
            } else {
                tmp = twoTurnTen(birth);
            }
            exp[i] = tmp;
            log[tmp] = i;
            i++;
            if (i > 8) {
                break;
            }
        }
        for (; i < exp.length; i++) {
            birth = new int[i + 1];
            for (int j : birth) {
                j = 0;
            }
            birth[i] = 1;
            int[] remainder = ModelPolynthicInverseAlgorithm.WithResidualDivider(birth.length - divisor.length, 2,
                    birth, divisor, new int[birth.length - divisor.length + 1]);
            tmp = twoTurnTen(remainder);
            exp[i] = tmp;
            log[tmp] = i;
        }
        exp[exp.length - 1] = 0;
    }

    /**
     * 二进制数组转十进制
     * @param twoNum 二进制数组
     * @return 十进制数值
     */
    private static int twoTurnTen(int[] twoNum) {
        int t = 1;
        int ten = 0;
        for (int i = 0; i < twoNum.length; i++) {
            if (twoNum[i] != 0)
                ten += t;
            t = t << 1;
        }
        return ten;
    }

    /**
     * 十进制数转二进制数组
     * @param tenNum 十进制数值
     * @return 二进制数组
     */
    private static int[] tenTurnTwo(int tenNum) {
        int index = 1;
        for (; index <= 8; index++) {
            if (Math.pow(2, index) > tenNum)
                break;
        }
        int[] two = new int[index];
        for (int i = 0; i < index; i++) {
            two[i] = tenNum % 2;
            tenNum /= 2;
        }
        return two;
    }

    /**
     * GF(2^8)域上元素乘法运算，调用实验三的方法
     * @param polynthic1 多项式1
     * @param polynthic2 多项式2

    private static int[] multiply(int[] polynthic1, int[] polynthic2, int[] polynthic3) {
        int[] multiplication = ModelPolynthicInverseAlgorithm.Multiplication(2, polynthic1, polynthic2);
        return ModelPolynthicInverseAlgorithm.div(2, multiplication, polynthic3, new int[1]);
    }

    /**
     * 把多项式数组转换成String类型的多项式
     * @param ans 要转换的多项式数组
     * @return String类型的多项式
     */
    private static String intArrayToString(int[] ans) {
        if (ans.length == 0) {
            return "0";
        }
        StringBuffer ansSb = new StringBuffer();
        for (int i = 0; i < ans.length; i++) {
            if (i == 0 && ans[i] != 0) {
                ansSb.append(1 + "+");
            } else if (i == 1 && ans[i] != 0) {
                ansSb.append("X" + "+");
            } else if (ans[i] == 1) {
                ansSb.append("X^" + i + "+");
            } else if (ans[i] != 0) {
                ansSb.append(ans[i] + "X^" + i + "+");
            }
        }
        return ansSb.length() - 1 > 0 ? ansSb.deleteCharAt(ansSb.length() - 1).toString() : ansSb.toString();
    }
}

实验原理：


public class Int {

    public int[] v;

    public Int() {
    };

    public Int(int[] v) {
        this.v = v;
    }

}

public class ModelPolynthicInverseAlgorithm {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入多项式计算数域Zp的p值：");
        int p = sc.nextInt();
        int[] polynthic1 = { 1, 1, 1, 1, 1, 1 }; // 1+X+X^2+X^3+X^4+X^5
        int[] polynthic2 = { 1, 1, 1, 0, 1 }; // 1+X+X^2+X^4

        System.out.println("选取两个多项式：" + intArrayToString(polynthic1) + " 和 " + intArrayToString(polynthic2));

        System.out.println("******");
        int[] mulAns = Multiplication(p, polynthic1, polynthic2);
        System.out.println("多项式乘法：" + intArrayToString(mulAns));

        System.out.println("******");
        int[] quotient = new int[polynthic1.length - polynthic2.length + 1]; // 商
        for (int quo : quotient) {
            quo = 0;
        }
        int[] remainder = WithResidualDivider(polynthic1.length - polynthic2.length, p, polynthic1, polynthic2,
                quotient);
        System.out.println("多项式带余除法的商：" + intArrayToString(quotient) + "，余数：" + intArrayToString(remainder));

        System.out.println("******");
        Int x = new Int();
        Int y = new Int();
        int[] gcd = Gcd(p, polynthic1, polynthic2, x, y);
        System.out.println("最大公因式为: " + intArrayToString(gcd));
        System.out.println("最大公因式可以表示为：" + intArrayToString(gcd) + " = " + "(" + intArrayToString(polynthic1) + ")"
                + "*" + "(" + intArrayToString(x.v) + ")" + "+" + "(" + intArrayToString(polynthic2) + ")" + "*" + "("
                + intArrayToString(y.v) + ")");
    }

    /**
     * 求乘法逆元
     * @param p 域
     * @param polynthic1 多项式1
     * @param polynthic2 多项式2
     * @param x 逆元1
     * @param y 逆元2
     * @return
     */
    public static int[] Gcd(int p, int[] polynthic1, int[] polynthic2, Int x, Int y) {
        int[] po1 = polynthic1.clone();
        int[] po2 = polynthic2.clone();
        // 调用递归函数
        return exGcd(p, po1, po2, x, y);
    }

    /**
     * 扩展的欧几里得方法求乘法逆元
     * @param p 域
     * @param polynthic1 多项式1
     * @param polynthic2 多项式2
     * @param x 逆元1
     * @param y 逆元2
     * @return
     */
    public static int[] exGcd(int p, int[] polynthic1, int[] polynthic2, Int x, Int y) {
        if (polynthic2.length == 0) {
            int[] X = { 1 };
            int[] Y = { 0 };
            x.v = X;
            y.v = Y;
            return polynthic1;
        }
        int[] quotient = new int[1];
        if (polynthic1.length - polynthic2.length > 0) {
            quotient = new int[polynthic1.length - polynthic2.length + 1];
        }
        int[] remainder = div(p, polynthic1, polynthic2, quotient);
        if (polynthic1.length == polynthic2.length) {
            quotient[0] = 1;
        } else if (polynthic1.length - polynthic2.length < 0) {
            quotient[0] = 0;
        }
        int[] gcd = exGcd(p, polynthic2, remainder, x, y);
        int[] c = y.v;
        y.v = Subtraction(p, x.v, Multiplication(p, quotient, y.v));
        x.v = c;
        return gcd;
    }

    /**
     * 多项式减法
     * @param subtracted 被减数
     * @param subtractor 减数
     * @return 减法结果数组
     */
    private static int[] Subtraction(int p, int[] subtracted, int[] subtractor) {
        if (subtracted.length < subtractor.length) {
            int[] temp = subtracted;
            subtracted = subtractor;
            subtractor = temp;
        }
        for (int i = 0, j = 0; i < subtracted.length && j < subtractor.length; i++, j++) {
            if (subtracted[i] - subtractor[j] >= 0) {
                subtracted[i] = subtracted[i] - subtractor[j];
            } else {
                subtracted[i] = p + subtracted[i] - subtractor[j];
            }
        }
        return subtracted;
    }

    /**
     * 多项式除法
     * @param p 域
     * @param remainder 被除数
     * @param divisor 除数
     * @param quotient 商
     * @return 余数
     */
    public static int[] div(int p, int[] remainder, int[] divisor, int[] quotient) {
        if (remainder.length == divisor.length) {
            return add(p, remainder, divisor);
        } else if (remainder.length - divisor.length > 0) {
            return WithResidualDivider(remainder.length - divisor.length, p, remainder, divisor, quotient);
        }
        return remainder;
    }

    /**
     * 多项式加法
     * @param p 域
     * @param polynthic1 多项式1
     * @param polynthic2 多项式2
     * @return
     */
    private static int[] add(int p, int[] polynthic1, int[] polynthic2) {
        int[] po1 = polynthic1.clone();
        int[] po2 = polynthic2.clone();
        if (po1.length < po2.length) {
            int[] temp = po1;
            po1 = po2;
            po1 = temp;
        }
        for (int i = 0; i < po2.length; i++) {
            if (po2[i] != 0) {
                po1[i] = (po1[i] + po2[i]) % p;
            }
        }
        return po1;
    }

    /**
     * 递归解决多项式除法
     * @param p 域
     * @param remainder 余数/被除数
     * @param divisor 除数（不变）
     * @param quotient 商
     */
    public static int[] WithResidualDivider(int len, int p, int[] remainder, int[] divisor, int[] quotient) {
        if (len < 0) {
            return remainder;
        }
        quotient[len] = Math.abs(remainder[remainder.length - 1]);
        for (int i = 0; i < divisor.length; i++) {
            if (divisor[i] != 0) {
                remainder[i + len] = Math.abs((remainder[i + len] - divisor[i]) % p);
            }
        }
        int index = remainder.length - 1;
        for (; index >= 0 && remainder[index] == 0; index--)
            ;
        int[] newReminder = new int[index + 1];
        for (int i = newReminder.length - 1; i >= 0 && index >= 0; i--, index--) {
            newReminder[i] = remainder[index];
        }
        return WithResidualDivider(newReminder.length - divisor.length, p, newReminder, divisor, quotient);
    }

    /**
     * 多项式乘法
     * @param p 域
     * @param polynthic1 多项式1
     * @param polynthic2 多项式2
     * @return 多项式乘法结果
     */
    public static int[] Multiplication(int p, int[] polynthic1, int[] polynthic2) {
        int len = polynthic1.length + polynthic2.length;
        int[] ans = new int[len - 1];
        for (int i : ans) {
            i = 0;
        }
        for (int i = 0; i < polynthic1.length; i++) {
            for (int j = 0; j < polynthic2.length; j++) {
                if (polynthic1[i] != 0 && polynthic2[j] != 0) {
                    ans[i + j] += polynthic1[i] * polynthic2[j];
                    ans[i + j] = ans[i + j] % p;
                }
            }
        }
        return ans;
    }

    /**
     * 把多项式数组转换成String类型的多项式
     * @param ans 要转换的多项式数组
     * @return String类型的多项式
     */
    private static String intArrayToString(int[] ans) {
        if (ans.length == 0) {
            return "0";
        }
        StringBuffer ansSb = new StringBuffer();
        for (int i = 0; i < ans.length; i++) {
            if (i == 0 && ans[i] != 0) {
                ansSb.append(1 + "+");
            } else if (i == 1 && ans[i] != 0) {
                ansSb.append("X" + "+");
            } else if (ans[i] == 1) {
                ansSb.append("X^" + i + "+");
            } else if (ans[i] != 0) {
                ansSb.append(ans[i] + "X^" + i + "+");
            }
        }
        return ansSb.length() - 1 > 0 ? ansSb.deleteCharAt(ansSb.length() - 1).toString() : ansSb.toString();
    }
}

模指数运算的实现（Java）

（1） 掌握常用的模指数算法的实现方法，即：求ae mod m 的值

（2） 算法采用重复平方乘的方法实现

（3）要求m的大小为512比特~1024比特，a和e可以随机产生

实验原理：
多精度整数除法
重复平方乘算法

代码如下：

public class Square {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入模m:");
        int m = sc.nextInt(); // 模
        int e = getRandom(); // 指数
        int b = getRandom(); // 底数

        String twoNum = turnTwo(e); // 指数转为二进制
        int r = modExponentialOperations(b, m, twoNum);
        System.out.println(b + "^" + e + "mod" + m + "=" + r);
    }

    /**
     * 生成指数e或者底数b
     * @return [1,100]随机数
     */
    private static int getRandom() {
        return (int) (Math.random() * 100) + 1;
    }

    /**
     * 模指数运算
     * @param b 底数
     * @param m 模
     * @param twoNum 指数的二进制
     * @return 余数
     */
    public static int modExponentialOperations(int b, int m, String twoNum) {
        int r = 1;
        for (int i = twoNum.length() - 1; i >= 0; i--) {
            if (twoNum.charAt(i) == '1') {
                r = (r * b) % m;
                b = (b * b) % m;
            } else {
                b = (b * b) % m;
            }
        }

        return r;
    }

    /**
     * 十进制转二进制
     * @param tenNum 十进制数
     * @return 二进制数
     */
    private static String turnTwo(int tenNum) {
        String str = "";
        while (tenNum != 0) {
            str = tenNum % 2 + str;
            tenNum /= 2;
        }
        return str;
    }

}

public class MultiPrecisionIntegerOperation {

    public static void main(String[] args) {
        String bigNum1 = "10101"; // 16+4+1=21
        String bigNum2 = "11"; // 3

        // 十进制转二进制：
        int tenNum1 = 20;
        int tenNum2 = 4;
        bigNum1 = turnTwo(tenNum1);
        bigNum2 = turnTwo(tenNum2);
        System.out.println(bigNum1 + "(" + tenNum1 + ")" + "与" + bigNum2 + "(" + tenNum2 + ")");

        String add = bigAdd(bigNum1, bigNum2);
        String sub = bigSubtracte(bigNum1, bigNum2);
        String mul = bigMultiplicate(bigNum1, bigNum2);
        String div = bigDivision(bigNum1, bigNum2);

        System.out.println(bigNum1 + "+" + bigNum2 + "=" + turnTen(Integer.parseInt(add)));// 24
        System.out.println(bigNum1 + "-" + bigNum2 + "=" + turnTen(Integer.parseInt(sub)));// 18
        System.out.println(bigNum1 + "*" + bigNum2 + "=" + turnTen(Integer.parseInt(mul)));// 63
        System.out.println(bigNum1 + "/" + bigNum2 + "=" + turnTen(Integer.parseInt(div)));// 7

    }

    /**
     * 二进制转十进制
     * @param twoNum 二进制数
     * @return 十进制数
     */
    private static String turnTen(int twoNum) {
        int decimal = 0;
        int p = 0;
        while (true) {
            if (twoNum == 0) {
                break;
            } else {
                int temp = twoNum % 10;
                decimal += temp * Math.pow(2, p);
                twoNum = twoNum / 10;
                p++;
            }
        }
        return String.valueOf(decimal);
    }

    /**
     * 十进制转二进制
     * @param tenNum 十进制数
     * @return 二进制数
     */
    private static String turnTwo(int tenNum) {
        String str = "";
        while (tenNum != 0) {
            str = tenNum % 2 + str;
            tenNum /= 2;
        }
        return str;
    }

    /**
     * 二进制除法
     * @param bigNum1 二进制数
     * @param bigNum2 二进制数
     * @return 商
     */
    private static String bigDivision(String bigNum1, String bigNum2) {
        int[] quotient = new int[bigNum1.length()];
        StringBuffer ans = new StringBuffer(); // 商
        doDivision(0, quotient, bigNum1, bigNum2);
        for (int i = 0; i < bigNum1.length(); i++) {
            ans.append(quotient[i]);
        }
        return ans.toString();
    }

    /**
     * 递归做除法
     * @param index 商的下标
     * @param quotient 商数组
     * @param remainder 被除数/余数
     * @param divisor 除数
     */
    private static void doDivision(int index, int[] quotient, String remainder, String divisor) {
        // 返回判断条件：余数/被除数小于除数
        if (isSmall(remainder, divisor)) {
            return;
        }

        int end = divisor.length();
        // 补零
        for (; index + remainder.length() - end + 1 < quotient.length; index++) {
            quotient[index] = 0;
        }

        String newRemainder = remainder.substring(0, end);
        while (isSmall(newRemainder, divisor)) {
            if (index < quotient.length - remainder.length() + end) {
                quotient[index] = 0;
                index++;
            }
            newRemainder = remainder.substring(0, ++end);
        }

        newRemainder = bigSubtracte(newRemainder, divisor);
        quotient[index++] = 1;

        // 新余数
        StringBuffer sbNewRemainder = new StringBuffer(newRemainder);
        sbNewRemainder.append(remainder.substring(end, remainder.length()));
        doDivision(index, quotient, sbNewRemainder.toString(), divisor);
    }

    /**
     * 判断二进制数大小
     * @param remainder 余数
     * @param divisor 被除数
     * @return true/false
     */
    private static boolean isSmall(String remainder, String divisor) {
        if (remainder.length() < divisor.length())
            return true;
        if (Integer.parseInt(remainder) < Integer.parseInt(divisor))
            return true;
        return false;
    }

    /**
     * 二进制乘法
     * @param bigNum1 二进制数
     * @param bigNum2 二进制数
     * @return 积
     */
    private static String bigMultiplicate(String bigNum1, String bigNum2) {
        String ans = "0";
        StringBuilder sb = null;
        for (int index = bigNum2.length() - 1; index >= 0; index--) {
            if (bigNum2.charAt(index) == '1') {
                // 给低位补零
                sb = new StringBuilder(bigNum1);
                for (int i = 0; i < bigNum2.length() - 1 - index; i++) {
                    sb.append('0');
                }
                ans = bigAdd(sb.toString(), ans);
            }
        }
        return ans;
    }

    /**
     * 二进制减法
     * @param bigNum1 二进制数
     * @param bigNum2 二进制数
     * @return 差
     */
    private static String bigSubtracte(String bigNum1, String bigNum2) {
        StringBuffer sb = new StringBuffer();
        StringBuffer sbZero = new StringBuffer();
        for (int i = 0; i < (bigNum1.length() - bigNum2.length()); i++) {
            sbZero.append('0');
        }
        sbZero.append(bigNum2);
        String bigString = sbZero.toString();
        for (int i = 0; i < bigString.length(); i++) {
            if (bigString.charAt(i) == '0') {
                sb.append('1');
            } else {
                sb.append('0');
            }
        }
        String ans = bigAdd("1", sb.toString());
        ans = bigAdd(ans, bigNum1);
        if (ans.length() > bigNum1.length()) {
            ans = ans.substring(1, ans.length());
        }
        int index = 0;
        for (; index < ans.length(); index++) {
            if (ans.charAt(index) == '1')
                break;
        }
        if (index == ans.length())
            return "0";
        return index != 0 ? ans.substring(index, ans.length()) : ans;
    }

    /**
     * 二进制加法
     * @param bigNum1 二进制数
     * @param bigNum2 二进制数
     * @return 和
     */
    private static String bigAdd(String bigNum1, String bigNum2) {
        int len1 = bigNum1.length();
        int len2 = bigNum2.length();
        String s1 = bigNum1;
        String s2 = bigNum2;
        StringBuffer sb1 = new StringBuffer();
        // 先将位数较少的二进制高位补零
        if (len1 > len2) {
            for (int i = 0; i < (len1 - len2); i++) {
                sb1.append(0);
            }
            sb1.append(bigNum2);
            s1 = bigNum1;
            s2 = sb1.toString();
        } else if (len1 < len2) {
            for (int j = 0; j < (len2 - len1); j++) {
                sb1.append(0);
            }
            sb1.append(bigNum1);
            s1 = sb1.toString();
            s2 = bigNum2;
        }
        // 进位
        int flag = 0;
        StringBuffer sb2 = new StringBuffer();
        for (int z = s1.length() - 1; z >= 0; z--) {
            if ((s1.charAt(z) - '0') == 0 && (s2.charAt(z) - '0') == 0) {
                sb2.append(flag);
                flag = 0;
                continue;
            }
            if (((s1.charAt(z) - '0') == 0 && (s2.charAt(z) - '0') == 1 && flag == 0)
                    || ((s1.charAt(z) - '0') == 1 && (s2.charAt(z) - '0') == 0 && flag == 0)) {
                sb2.append(1);
                flag = 0;
                continue;
            }
            if (((s1.charAt(z) - '0') == 0 && (s2.charAt(z) - '0') == 1 && flag == 1)
                    || ((s1.charAt(z) - '0') == 1 && (s2.charAt(z) - '0') == 0 && flag == 1)) {
                sb2.append(0);
                flag = 1;
                continue;
            }
            if ((s1.charAt(z) - '0') == 1 && (s2.charAt(z) - '0') == 1 && flag == 0) {
                sb2.append(0);
                flag = 1;
                continue;
            }
            if ((s1.charAt(z) - '0') == 1 && (s2.charAt(z) - '0') == 1 && flag == 1) {
                sb2.append(1);
                flag = 1;
            }

        }
        if (flag == 1) {
            sb2.append(flag);
        }
        // 倒置
        sb2.reverse();
        return sb2.toString();
    }

简单移位加密算法-斯巴达手杖

斯巴达手杖是一种借助工具实现的移位加密方法，其基本原理是将书写信息的纸条（或羊皮）缠绕在特定的木棍上，然后自上到下写上明文信息，最后将字条从木棍上取下，构成明文的字母顺序就被打乱，当纸条重新被缠绕回相同的木棍时字母顺序被还原，从而可以读取明文信息。假定木棍的截面为一个正六边型，简单的示意如图1所示。

模拟斯巴达手杖的加密过程，可以看做将明文按列写入矩阵，然后按行输出到数组，通过一个逆过程可以重新得到明文。

    /**
     * 生成密钥
     * @param Plaintext 明文
     * @return 密钥
     */
    private static int getRandom(String Plaintext) {
        // 根据字符串的长短确定密钥的生成范围
        if (Plaintext.length() > 300) { // [10,30]
            return (int) (Math.random() * 30) + 10;
        } else if (Plaintext.length() > 10) { // [2,10]
            return (int) (Math.random() * 10) + 2;
        }
        return (int) (Math.random() * 3) + 2; // [2,3]
    }

    /**
     * 解密
     * @param password 密钥
     * @param ciphertext 密文
     * @return 明文
     */
    private static String decrypt(int password, String ciphertext) {
        Map<Integer, String> deMap = new HashMap<Integer, String>();
        // 计算列
        int num = ciphertext.length() / password; // 商
        int r = ciphertext.length() % password; // 余数
        boolean flag = r == 0 ? false : true; // true:有余数，false:无余数
        int size = r == 0 ? num : num + 1;
        StringBuilder sb = null;
        char[] ciphChar = ciphertext.toCharArray();

        // 先把密文按照行的方式输入第一行
        for (int i = 0; i < size; i++) {
            deMap.put(i % size, String.valueOf(ciphChar[i]));
        }

        // 把密文按照行的方式把剩下的输入
        // 要注意输入最后一列的时候，要分成两种情况
        for (int i = size, j = i; j < ciphChar.length; i++, j++) {
            // 有余数并且在最后一行
            if (flag && i % size == size - 1) {
                if (r - 1 > 0) { // 最后一行还能输入
                    r--;
                } else { // 不能输入了
                    j--;
                    continue;
                }
            }
            sb = new StringBuilder(deMap.get(i % size));
            deMap.put(i % size, sb.append(ciphChar[j]).toString());
        }

        // 遍历map，按照列的方式append到sb中
        Collection<String> values = deMap.values();
        sb = new StringBuilder();
        for (String str : values) {
            sb.append(str);
        }

        return sb.toString();
    }

    /**
     * 加密
     * @param password 密钥
     * @param plaintext 明文
     * @return 密文
     */
    private static String encrypt(int password, String plaintext) {
        // 相当于把ciMap变成一个i列password行的二维数组
        Map<Integer, String> ciMap = new HashMap<Integer, String>(); // key：第i列，value：密文
        int key = 0; // ciMap的key值
        StringBuilder sb = new StringBuilder();
        char[] ch = null; // 用于存放ciMap每一列的字符
        // 将明文按照列存放进map
        for (int i = 0; i < plaintext.length(); key++) {
            if (i + password >= plaintext.length()) {// 最后一行不够password个字符，直接截取到最后一个
                ciMap.put(key, plaintext.substring(i));
                break;
            }
            ciMap.put(key, plaintext.substring(i, i + password)); // 截取password个字符
            i += password;
        }

        // 从map按照行取出字符作为密文
        for (int i = 0; i < password; i++) {// 遍历行
            for (int index = 0; index < ciMap.size(); index++) {// 遍历列
                // 将第index列的第i个字符append
                ch = ciMap.get(index).toCharArray();
                if (i >= ch.length)
                    break;
                sb.append(ch[i]);
            }
        }

        return sb.toString();
    }

}

     * 解密
     * @param key 密钥
     * @param ciphertext 密文
     * @return 明文
     */
    private static String decrypt(int key, String ciphertext) {
        StringBuilder sb = new StringBuilder();
        char[] ch = ciphertext.toCharArray();
        for (int i = 0; i < ch.length; i++) {
            if (Character.isLowerCase(ch[i])) { // 如果是小写字母
                sb.append((char) ((ch[i] - 'a' - key + 26) % 26 + 'a'));
            } else if (Character.isUpperCase(ch[i])) { // 如果是大写字母
                sb.append((char) ((ch[i] - 'A' - key + 26) % 26 + 'A'));
            } else { // 其余直接append
                sb.append(ch[i]);
            }
        }
        return sb.toString();
    }

            if (Character.isLowerCase(ch[i])) { // 如果是小写字母

            } else { // 其余直接append

            }
        }

#include <string.h>

int Find(HashTable* table, const char* key, int value, long keyhash)
{

typedef int VertexType;
typedef enum{
    DG, UDG
}GraphType;

ypedef struct ArcNode
{
    int adjvex;
    InfoPtr *info;
    struct ArcNode *nextarc;
 }ArcNode;

typedef struct VNode
{
    VertexType data;
    ArcNode *firstarc;
}VNode;

typedef struct
{
    VNode vertex[MAX_VERTEX_NUM];
    int vexnum, arcnum;
    GraphType type;
}ListGraph;

int locate_vertex(ListGraph* G, VertexType v); //返回顶点 v 在vertex数组中的下标，如果v不存在，返回-1
bool insert_vertex(ListGraph *G, VertexType v);
bool insert_arc(ListGraph *G, VertexType v, VertexType w);

当成功删除顶点或边时，函数返回true，否则（如顶点或边不存在、删除边时顶点v或w不存在）返回false。

bool del_vertex(ListGraph* G, VertexType v)
{
    int i, V = locate_vertex(G, v);
    if (V == -1) //检查是否存在该节点
        return false; 

    //先删除从该节点出发的边和该节点
    while (G->vertex[V].firstarc) { 
        ArcNode* P = G->vertex[V].firstarc;
        if (P->nextarc) { //先free表头结点后面的
            ArcNode* temp = P->nextarc;
            P->nextarc = temp->nextarc;
            free(temp);
        } else {
            free(P); //free表头结点
            G->vertex[V].firstarc = NULL;
        }
        G->arcnum--; //边的数量-1
    }
    G->vexnum--; //结点的数量-1
    for (i = V; i < G->vexnum; i++) { //表头结点中，后面的向前移动
        G->vertex[i] = G->vertex[i + 1];
    }

    //再删除到该节点的边
    for (i = 0; i < G->vexnum; i++) {
        ArcNode *P = G->vertex[i].firstarc, *pNode = NULL;
        ArcNode* temp; //存储要被删掉的结点
        while (P) {
            if (V == P->adjvex) { //P的下个结点是V
                if (!pNode) { //P是表头结点
                    temp = G->vertex[i].firstarc;
                    G->vertex[i].firstarc = P->nextarc;
                }
                else {
                    pNode->nextarc = P->nextarc;
                    temp = P;
                }
                P = P->nextarc;
                free(temp);
                G->arcnum--;
            } else {
                pNode = P;
                P = P->nextarc;
            }
        }
    }
    return true;
}
