模指数运算的实现（Java）

发表于 2020-12-20 | 分类于 公钥密码学

（1） 掌握常用的模指数算法的实现方法，即：求ae mod m 的值

（2） 算法采用重复平方乘的方法实现

（3）要求m的大小为512比特~1024比特，a和e可以随机产生

实验原理：
多精度整数除法
重复平方乘算法

代码如下：

public class Square {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入模m:");
        int m = sc.nextInt(); // 模
        int e = getRandom(); // 指数
        int b = getRandom(); // 底数

        String twoNum = turnTwo(e); // 指数转为二进制
        int r = modExponentialOperations(b, m, twoNum);
        System.out.println(b + "^" + e + "mod" + m + "=" + r);
    }

    /**
     * 生成指数e或者底数b
     * @return [1,100]随机数
     */
    private static int getRandom() {
        return (int) (Math.random() * 100) + 1;
    }

    /**
     * 模指数运算
     * @param b 底数
     * @param m 模
     * @param twoNum 指数的二进制
     * @return 余数
     */
    public static int modExponentialOperations(int b, int m, String twoNum) {
        int r = 1;
        for (int i = twoNum.length() - 1; i >= 0; i--) {
            if (twoNum.charAt(i) == '1') {
                r = (r * b) % m;
                b = (b * b) % m;
            } else {
                b = (b * b) % m;
            }
        }

        return r;
    }

    /**
     * 十进制转二进制
     * @param tenNum 十进制数
     * @return 二进制数
     */
    private static String turnTwo(int tenNum) {
        String str = "";
        while (tenNum != 0) {
            str = tenNum % 2 + str;
            tenNum /= 2;
        }
        return str;
    }

}
